Задача 1. Дан массив отрицательных и положительных чисел. Произвести сложение всех четных положительных чисел , и умножение всех нечетных положительных чисел ,после чего возвести сумму первых ,в степень произведения вторых. Если полученное число будет больше 777,то делим это число на сумму модулей всех отрицательных чисел массива. Выводим конечный результат. 

        var a =[1,2,3,4,3,-1,-2,-3,-4,-5,-6];
    
            var ch=0;
            var nch=1;
            var mod=1;
    
    
        for(let i=0;i<a.length;i++)
        {
            if(a[i]>=0 && a[i]%2==0)
            ch+=a[i];
    
            if(a[i]>=0 && a[i]%2!=0)
            nch*=a[i];
    
            if(a[i]<0)
            mod+=a[i]*-1;
        }
    
        var k = ch;
    
        for(let i=1;i<nch;i++)
        {
            k*=ch;
        }
    
        if(k>777)
        k/=mod;
            
        document.write(Math.round(k,5));
